Lexical Syntax

Variables
ID     -> string
NUM    -> string of 0-9

Parenthesis
LPAREN -> (
RPAREN -> )
LBRACE -> {
RBRACE -> }
LBRACK -> [
RBRACK -> ]
QUOTE  -> "

Operators
BECOMES-> =
DOT    -> .
COMMA  -> ,
PLUS   -> +
MINUS  -> -
STAR   -> *
SLASH  -> /
SEMI   -> ;

Comparators
EQ     -> equals
AND    -> and
OR     -> or
NOT    -> not
LT     -> <
GT     -> >
LE     -> <=
GE     -> =>

Control flow
IF     -> if
ELSE   -> else
WHILE  -> while
RETURN -> return

Built-in functions
PRINT  -> print
ADDR   -> addr
AT     -> at
